#include<iostream>

using namespace std;

int main()
{
     int n;
     cin>>n;
     for(int i=1;i<=3;i++){
         for(int j=1;j<=n;j++){
             if((i+j)%4==0){     //by adding row and column we found that it is always divisible by 4(1st condition)
                 cout<<"*";
             }
             
            else if(j%4==0&&i==2){      // (secod condition)
                 cout<<"*";
             }
             else
             cout<<" ";
         }
         cout<<endl;
         
     }
}
